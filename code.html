Common Character Count

//Comment
function commonCharacterCount(s1, s2) {
    var ret = 0;
    for(var idx = 0; idx < s1.length; idx++) {
        for(var xdx = 0; xdx < s2.length; xdx++) {
            if(s1[idx] == s2[xdx]) {
                ret++;
                s2 = s2.replace(s2[xdx], '');
                break;
            }
        }
    }
    return ret;
}

Sort By Height

//Comment
function check(v) {
    return v === -1;
}
function sortByHeight(a) {
    var newArr = [];
    var treeIdx = [];
    var oLength = a.length;
    if(a.some(check)) {
        for(var idx = 0; idx < a.length; idx++) {
            if(a[idx] === -1) {
                treeIdx.push(idx);
            }
        }
        a.sort(function(a, b) {return a-b});
        for(var idx = 0; idx < treeIdx.length; idx++) {
            a.shift();
        }
        for(var idx = 0; idx < oLength; idx++) {
            if(idx === treeIdx[0]) {
                newArr.push(-1);
                treeIdx.shift();
            } else {
                newArr.push(a[0]);
                a.shift();
            }
        }
        return newArr;
    } else {
        return a.sort(function(a, b) {return a-b});
    }
}

function chunkArrayInGroups(arr, size) {
  var SL = arr.length;
  retArr = [];
  for(var i = 0; i < SL; i+=size) {
    retArr.push(arr.slice(0,size));
    for(var x = 0; x < size; x++) {
      arr.shift();
    }
  }
  return retArr;
}
